create table Person(
    id int primary key generated by default as identity ,
    unique_identity int8 unique not null ,
    first_name varchar(100) not null ,
    second_name varchar(100) not null ,
    email varchar(100) not null  unique ,
    address_permanent_residence varchar(200) not null,
    correspondence_address varchar(200)
);


create table address(

    id int primary key generated by default as identity ,
    zip_code int not null ,
    street_name varchar(200) not null ,
    house_number int not null,
    address_type int not null,
    village_name varchar(200) not null

);

create table contract(
    id int primary key generated by default as identity ,
    person_id int references Person(id) on delete cascade,
    unique_number int8 unique not null ,
    date_registration varchar(100) not null

);

create table property_contract(
    id int generated by default as identity ,
    contract_id int unique references contract(id) on DELETE cascade ,
    address_id int references address(id) ,
    property_type int not null ,
    cost_property float8 not null ,
    created_at varchar(100) not null

);


create table travel_contract(
    id int primary key generated by default as identity ,
    date_beginning varchar(100) not null ,
    date_end varchar(100) not null ,
    liability_damage boolean not null ,
    accident_insure boolean not null
);
